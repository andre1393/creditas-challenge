loan_transaction['year'] = loan_transaction['End Date'].apply(lambda x: x.split('/')[2])
grp = loan_transaction.groupby(['Country Name', 'year']).sum().reset_index()
grp.rename(columns = {'Country Name': 'country'}, inplace = True)
hdi.loc[:,'year'] = hdi['year'].apply(str)
merge = pd.merge(grp, hdi, on = ['country', 'year'], how = 'outer').dropna(subset = ['country', 'value', 'year', 'Amount (USD)'])
correlations = []
for country in merge['country'].unique():
    df_country = merge[merge['country'] == country]
    
    # somente calcula a correlacao se tiver mais de tres ocorrencias
    if len(df_country > 3):
        # utiliza a quantia cumulativa
        merge['Amount (USD)'] = merge['Amount (USD)'].cumsum() 
        cor = df_country.corr().loc['Amount (USD)', 'value']
        if not np.isnan(cor):
            correlations.append((country, cor))

print(correlations)
